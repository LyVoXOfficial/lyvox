## Currency & Number Formatting

**Helper функции:**
```typescript
// apps/web/src/lib/i18n/formatCurrency.ts
export function formatCurrency(
  amount: number,
  currency: string = 'EUR',
  locale: string
): string {
  return new Intl.NumberFormat(locale, {
    style: 'currency',
    currency,
    minimumFractionDigits: 0,
    maximumFractionDigits: 0
  }).format(amount);
}

// Примеры:
// formatCurrency(4900, 'EUR', 'nl') → "4.900 €"
// formatCurrency(4900, 'EUR', 'fr') → "4 900 €"
// formatCurrency(4900, 'RUB', 'ru') → "4 900 ₽"
```

**Форматирование дат:**
```typescript
// apps/web/src/lib/i18n/formatDate.ts
export function formatDate(
  date: Date | string,
  locale: string,
  options?: Intl.DateTimeFormatOptions
): string {
  return new Intl.DateTimeFormat(locale, {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
    ...options
  }).format(new Date(date));
}
```

## Language Switcher

**Компонент:** `apps/web/src/components/LanguageSwitcher.tsx`

**Функциональность:**
- Dropdown в header
- Сохранение выбора в cookie
- URL обновление: `/?lang=nl` или `/nl/`

**Implementation:**
```typescript
function LanguageSwitcher() {
  const [locale, setLocale] = useLocale();
  
  const handleChange = (newLocale: string) => {
    setLocale(newLocale);
    // Сохранение в cookie
    document.cookie = `NEXT_LOCALE=${newLocale}; path=/; max-age=31536000`;
    // Обновление URL или reload
    window.location.href = `/${newLocale}${window.location.pathname.replace(/^\/(nl|fr|en|ru)/, '')}`;
  };
  
  return (
    <Select value={locale} onValueChange={handleChange}>
      <SelectItem value="nl">Nederlands</SelectItem>
      <SelectItem value="fr">Français</SelectItem>
      <SelectItem value="en">English</SelectItem>
      <SelectItem value="ru">Русский</SelectItem>
    </Select>
  );
}
```

## Category Localization

**Приоритет локализации:**
1. Запрошенная локализация (из cookie/URL)
2. Fallback на английский (`name_en`)
3. Fallback на русский (`name_ru`)

**Helper:**
```typescript
function getCategoryName(category: Category, locale: string): string {
  const localeMap: Record<string, keyof Category> = {
    nl: 'name_nl',
    fr: 'name_fr',
    en: 'name_en',
    ru: 'name_ru'
  };
  
  const field = localeMap[locale] || 'name_en';
  return category[field] || category.name_en || category.name_ru || category.slug;
}
```

**Helper для выбора шаблона:**
```typescript
async function getEmailTemplate(
  templateName: string,
  locale: string
): Promise<React.ComponentType<Props>> {
  try {
    return await import(`./templates/${templateName}/${locale}`);
  } catch {
    // Fallback на английский
    return await import(`./templates/${templateName}/en`);
  }
}
```

## API Errors Localization

**Структура:**
```typescript
// apps/web/src/lib/i18n/apiErrors.ts
const errorMessages: Record<string, Record<string, string>> = {
  nl: {
    'UNAUTHENTICATED': 'U moet ingelogd zijn om deze actie uit te voeren',
    'NOT_FOUND': 'Niet gevonden',
    // ...
  },
  fr: {
    'UNAUTHENTICATED': 'Vous devez être connecté pour effectuer cette action',
    'NOT_FOUND': 'Non trouvé',
    // ...
  },
  // ...
};

export function getErrorMessage(code: string, locale: string): string {
  return errorMessages[locale]?.[code] || errorMessages['en'][code] || code;
}
```
